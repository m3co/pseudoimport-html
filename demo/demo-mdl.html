In order to import an HTML document from an URI, use the class .mdl-fragment in conjunction with src attribute.

demo-mdl.html 

<div id="fragment-mdl"
     src="/trunk/demo-mdl-fragment.html"
     class="mdl-fragment">
</div>
demo-mdl-fragment.html 

<!doctype html>
<div>Fragment 1</div>
<script>
  'use strict';
  document.querySelector('#fragment-mdl').addEventListener('load', function (e) {
    console.log('fragment loaded detected by inline script');
  });
</script>
